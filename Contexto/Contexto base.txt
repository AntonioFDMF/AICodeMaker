import pyautogui
from datetime import datetime as dt, time
from time import sleep
import subprocess


# Espera até o momento indicado
def alarm(h, m, s):
    momento = time(h, m, s).hour * 3600 + time(h, m, s).minute * 60 + time(h, m, s).second
    agora = dt.now().hour * 3600 + dt.now().minute * 60 + dt.now().second
    tempo = momento - agora

    sleep(tempo)


# Espera pelo período indicado
def timer(h, m, s):
    tempo = time(h, m, s).hour * 3600 + time(h, m, s).minute * 60 + time(h, m, s).second

    sleep(tempo)


# Mantém o computador ligado até o momento indicado

    agora = dt.now().hour * 3600 + dt.now().minute * 60 + dt.now().second
    tempo = momento - agora

    if tempo <= 0:
